/**
 * Set or get theme
 * 
 * @param string $new_theme
 * @return string
 */
function theme ($new_theme = '') {
    static $theme = 'default';

    if ($new_theme) {
        $theme = $new_theme;
    }

    if (file_exists($functions = view_path($theme, 'functions'))) {
        require_once $functions;
    }

    return $theme;
}
